(defstruct our help options)

(defmethod print-object ((o our) s)
  (format s "~a~%~%OPTIONS:~%" (our-help o))
  (dolist (x (our-options o)) (print-object  (cdr x) s )))

(defvar *config* (make-our))

(defmacro $ (x) `(cdr (assoc ',x (our-options *config*))))
